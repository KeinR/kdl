```
S* = PROGRAM
PROGRAM = ST_LIST
ST_LIST= PST ST_LIST
ST_LIST= PST
PST = ( ST )
PST = ( MARK ST )
PST = COMMENT
COMMENT = #.*\n
ST = COMP ? EXEC

COMP = COND_LIST
// Empty conditional is always true
COMP = <empty>

// So that the top level doesn't require parenthesies
// , = and, ; = or
COND_LIST = ( COND_LIST ) , COND_LIST
COND_LIST = ! ( COND_LIST ) , COND_LIST
COND_LIST = ( COND_LIST ) ; COND_LIST
COND_LIST = ! ( COND_LIST ) ; COND_LIST
COND_LIST = OP , COND_LIST
COND_LIST = OP , COND_LIST
COND_LIST = ! OP ; COND_LIST
COND_LIST = ! OP ; COND_LIST
COND_LIST = OP
COND_LIST = ! OP

OP = ( OP )
OP = VALUE = VALUE
OP = VALUE != VALUE
OP = VALUE < VALUE
OP = VALUE > VALUE
OP = VALUE <= VALUE
OP = VALUE >= VALUE
OP = VALUE
VALUE = ( VALUE )
VALUE = VALUE + VALUE
VALUE = VALUE - VALUE
VALUE = VALUE / VALUE
VALUE = VALUE * VALUE
VALUE = LITERAL
VALUE = VARIABLE
VARIABLE = {>?[^{}^]+}
VARIABLE = WORD
LITERAL = NUMBER
LITERAL = STRING
// Our dear conventional strings
// Though still hardly conventional
STRING = [ [^[]]+ ]
NUMBER = \d+(\.\d*)?%?
WORD_LIST = WORD WORD_LIST
WORD_LIST = WORD
WORD = [_a-zA-Z_][_a-zA-Z_0-9]*

MARK = JUMPER WORD_LIST :
// This sets global context
MARK = JUMPER :
JUMPER = <empty>
JUMPER = ^ JUMPER

EXEC = ORDER :: AFTER
EXEC = :: AFTER
EXEC = ORDER

ORDER = VERB
ORDER = VERB PARAM_LIST

PARAM_LIST = PARAM PARAM_LIST
PARAM_LIST = PARAM

PARAM = VALUE

VERB = WORD
```
