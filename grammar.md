COND_LIST_TL = MARK COND_LIST
COND_LIST_TL = COND_LIST
// , = and, ; = or
COND_LIST = ( COND_LIST ) , COND_LIST
COND_LIST = ! ( COND_LIST ) , COND_LIST

COND_LIST = OP , COND_LIST
COND_LIST = ! OP ; COND_LIST
COND_LIST = ! OP ; COND_LIST
COND_LIST = OP
COND_LIST = ! OP

OP = VALUE = VALUE

OP = VALUE < VALUE
OP = VALUE > VALUE
OP = VALUE <= VALUE
OP = VALUE >= VALUE
OP = VALUE
VALUE = ( MARK VALUE )
VALUE = ( VALUE )
VALUE = VALUE + VALUE
VALUE = VALUE - VALUE
VALUE = VALUE / VALUE
VALUE = VALUE * VALUE
VALUE = LITERAL
VALUE = VARIABLE

LITERAL = NUMBER

// Though still hardly conventional
LITERAL = [ WORD_OR_NUMBER_LIST ]

STRING = ( WORD_LIST )

WORD_LIST = WORD WORD_LIST
WORD_LIST = WORD

SET = :: AFTER

PARAM_LIST = PARAM
